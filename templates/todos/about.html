POST Request
fetch('/api/todos?api_key=apikey', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({
            title: 'New Task',
            description: 'Description for new task',
        }),
    })
    .then((response) => response.json())
    .then((json) => console.log(json));
